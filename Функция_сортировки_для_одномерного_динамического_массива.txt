void swap(int*a,int*b){
int buf;                                                //функция обмена значений переменных
buf=*a;*a=*b;*b=buf;
}

void sort_po(int*a,int n){                            //функция сортировки простым обиеном
	int i,j,flag;
	flag=1;
	while (flag) {
		flag=0;
		for (i=0;i<n-1;i++)
			if (a[i]>a[i+1]) {
				swap (&a[i],&a[i+1]); flag=1;}
	}
}
